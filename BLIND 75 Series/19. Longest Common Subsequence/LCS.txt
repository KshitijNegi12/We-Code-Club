#include<bits/stdc++.h>
using namespace std;
int lcs(string s1,string s2, int i , int j)
{
    if(i==s1.length() || j==s2.length())
    {
        return 0;
    }
    int count =0;
    if(s1[i]==s2[j])
    {
        count=1+lcs(s1,s2,i+1,j+1);
    }
    else 
    {
        count = max(lcs(s1,s2,i+1,j),lcs(s1,s2,i,j+1));
    }
    return count;
}
int main()
{
    string s1,s2;
    cin>>s1>>s2;
    cout<<lcs(s1,s2,0,0);
}

Time Complexity: O(N)
Space Complexity: O(1)

class Solution {
public:
    int maxArea(vector<int>& height) {
        int i=0,j=height.size()-1,ans=0;
        while(i<j){
            ans=max(ans,min(height[i],height[j])*(j-i));
            if(height[i]>height[j]){
                j--;
            }
            else{
                i++;
            }
        }
        return ans;
    }
};
